     \StrBefore{#1}{+}[\RealPart]}{%
     \IfSubStr[2]{#1}{-}{%
          \StrBefore[2]{#1}{-}[\RealPart]}{%
          \StrBefore{#1}{-}[\RealPart]}%
     }
     \StrBehind{#1}{i}[\ImagPart]%

    $\makebox[\WidestRealNum][r]{$\RealPart$} + i\,\ImagPart$}{%
    $\makebox[\WidestRealNum][r]{$\RealPart$} - i\,\ImagPart$}
}%
